CREATE KEYSPACE accounts WITH replication = {'class': 'SimpleStrategy', 'replication_factor': '1'}  AND durable_writes = true;

CREATE TABLE IF NOT EXISTS accounts.facebook_users (
  id text PRIMARY KEY, 
  ndla_id text,
  first_name text,
  middle_name text,
  last_name text,
  email text,
  created timeuuid
);

CREATE TABLE IF NOT EXISTS accounts.google_users (
  id text PRIMARY KEY,
  ndla_id text,
  first_name text,
  middle_name text,
  last_name text,
  display_name text,
  etag text,
  object_type text,
  email text,
  verified boolean,
  created timeuuid
);

CREATE TABLE IF NOT EXISTS accounts.ndla_users (
  id text PRIMARY KEY,
  first_name text,
  middle_name text,
  last_name text,
  email text,
  created timeuuid,
  facebook_id text,
  google_id text, 
  twitter_id text
);


CREATE KEYSPACE states WITH replication = {'class': 'SimpleStrategy', 'replication_factor': '1'}  AND durable_writes = true;

CREATE TABLE IF NOT EXISTS accounts.state (
  id uuid PRIMARY KEY
);
